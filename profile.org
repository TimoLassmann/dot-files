  #+BEGIN_SRC sh
    OLDPATH=$PATH
    PATH=$HOME/bin

    for DIR in /opt/local/bin /opt/local/sbin /usr/local/bin /usr/local/sbin
    do
      if [ -d $DIR ]
      then
          PATH=$PATH:$DIR
      fi
    done

    PATH=$PATH:$OLDPATH
  #+END_SRC

  #+BEGIN_SRC sh 
    if which gls >/dev/null 2>&1
    then
        alias ls="gls --color"
        alias ll="gls --color -olhA"
    else
        alias ls="ls --color"
        alias ll="ls --color -olhA"
    fi
  #+END_SRC
